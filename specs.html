Trigger logic

function wigglePromptBar() {
  const card = document.querySelector(".prompt-bar-card");
  if (!card) return;
  card.classList.remove("is-wiggling");
  // Force reflow so the animation retriggers on rapid clicks.
  void card.offsetWidth;
  card.classList.add("is-wiggling");
  card.addEventListener("animationend",
    () => card.classList.remove("is-wiggling"),
    { once: true });
}

// Called when the send button is pressed with no text.
if (!askInput.value.trim()) {
  wigglePromptBar();
  askInput.focus();
}

Trigger logic

function openStarterCard(chip, card, header) {
  const chipRect = chip.getBoundingClientRect();
  const expandedWidth = 660;
  const expandedLeft = cardRect.left + (cardRect.width - expandedWidth) / 2;

  // 1. Start the card exactly on top of the chip.
  card.style.left = `${chipRect.left}px`;
  card.style.top = `${chipRect.top}px`;
  card.style.width = `${chipRect.width}px`;
  card.style.maxHeight = `${chipRect.height}px`;
  header.style.height = `${chipRect.height}px`;

  card.hidden = false;
  chip.style.opacity = '0';              // hide chip under card
  chip.setAttribute('aria-expanded', 'true');
  panelsBlur.classList.add('is-active');   // fade blur in

  // 2. Next frame, flip to the expanded geometry → transitions run.
  requestAnimationFrame(() => requestAnimationFrame(() => {
    card.classList.add('is-expanded');
    card.style.left = `${expandedLeft}px`;
    card.style.width = '660px';
    card.style.maxHeight = '320px';
  }));
}

function closeStarterCard(chip, card) {
  const chipRect = chip.getBoundingClientRect();
  card.classList.remove('is-expanded');   // reverse to chip footprint
  card.style.left = `${chipRect.left}px`;
  card.style.width = `${chipRect.width}px`;
  card.style.maxHeight = `${chipRect.height}px`;
  panelsBlur.classList.remove('is-active');
  card.addEventListener('transitionend', () => {
    card.hidden = true;
    chip.style.opacity = '';              // restore chip
  }, { once: true });
}

Loader (Warming up)

The loading state shown while Spotter reasons on a query.

Recommendation Do not use the literal “Warming up Spotter” text loader. Instead, use the moving gradient background — the same radiance wash documented in BG fadeout — as the loading indicator. The drifting gradient communicates “thinking” ambiently, without a literal label, and it transitions seamlessly into the answer state via the background fade-out (translateY −500px, opacity → 0, 5000ms ease-out).

BG fadeout

The radiance wash lives at the top of the surface and drifts continuously. When an answer loads, the whole background fades up and out — it translates toward the top while its opacity falls to zero, clearing the canvas for the answer. Press Play to watch the chronological flow: question → reasoning → answer, with the fade-out firing as the answer settles.

Fade-out on answer load

// Fires ~530ms after the answer chart fades in
// (480ms answer scroll + 50ms).
setTimeout(() => {
  animateBgTo(-500, 0, 5000);  // y = -500px, opacity 0%, over 5000ms
}, 480 + 50);

function animateBgTo(y, opacityPct, ms) {
  const easing = `${ms}ms cubic-bezier(0.0, 0.0, 0.2, 1)`; // ease-out
  bgRoot.style.transition = `opacity ${easing}, transform ${easing}`;

  // Overlay ::before/::after fade instantly (0ms) so they don't
  // lag the canvas as it moves and reveal a double gradient.
  root.style.setProperty('--bg-anim-duration', '0ms');
  homeMain.style.setProperty('--bg-overlay-opacity', opacityPct / 100);
  root.style.setProperty('--bg-anim-duration', `${ms}ms`);

  bgRoot.style.transform = `translateY(${y}px)`;  // drifts up → toward top
  bgRoot.style.opacity   = opacityPct / 100;       // → 0
}
